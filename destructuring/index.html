Destructuring

Destructuring is used to shorten and clean up our codes. Lets us take properties from an object, or values from an array. Dot notation, Assign properties to variables, destructuring the properties.

How to destructure the values of an Object 

        
            const {
                /* properties */
            } = objectName;

            // For example:
            const { property1, property2 } = objectName;

How to use the destructured Objects 

        
          const person = {
              firstName: 'Ola',
              lastName: 'Nordmann',
              country: 'Norway',
            };

            // Normal variable assignment to object properties with Dot notation:
            const firstName = person.firstName;
            const lastName = person.lastName;
            const country = person.country;

            // Destructuring the values(properties) instead:
            const { firstName, lastName, country } = person;

            // Using the destructured values:
            const greeting = `Hello, I am ${firstName} ${lastName} from ${country}.`;
            console.log(greeting);
            // Logs:
            // Hello, I am Ola Nordmann from Norway.

How to use it in a function 

        
            function greetPerson(personObject) {
              const { firstName, lastName } = personObject;
              console.log(`Hello ${firstName} ${lastName}.`);
            }

            // Destructure the properties from the object
            function greetPerson({ firstName, lastName }) {
              console.log(`Hello ${firstName} ${lastName}.`);
            }

            const person = {
              firstName: 'Ola',
              lastName: 'Nordmann',
            };

            greetPerson(person);

How to use destructuring in a Array

        
            // Get only people that has scored 5 or more:
            const people = [
              {
                firstName: 'Berit',
                lastName: 'Nilsen',
                score: 8,
              },
              {
                firstName: 'Lars',
                lastName: 'Nilsen',
                score: 9,
              },
              {
                firstName: 'Nina',
                lastName: 'Person',
                score: 4,
              }
            ];

            //Using .filter() the old way:
            const winnersOld = people.filter((currentItem) =>
            {
              if (currentItem.score >= 5) {
                return true;
              }
            });
            console.log(winnersOld);

            // Using .filter() the new way:
            const winnersNew = people.filter(({ score }) => {
              if (score >= 5) {
                return true;
              }
            });

            console.log(winnersNew);

Other example with arrays

        
            // example destructuring array with numbers
            //           index:    0   1   2   3   4   5
            const streetNumbers = [12, 30, 34, 50, 1, 67];

            const [x, y] = streetNumbers; // X=0 Y=1 in index

            console.log(x, y); // Log: 12, 30
        
    
        
            const names = ["Hesh", "Linda", "Alex", "Fariad", "Yasmin", "Aladdin", "Jenny"];

            const [nameX, nameY, nameZ, ...others] = names;

            console.log(nameX, nameY, nameZ); // Logs: Hesh Linda Alex
            console.log(...others); // Logs: Fariad Yasmin Aladdin Jenny
            console.log(others); // Logs: the others in an Array
        
    
        
            // Ignoring Values:
            const numbers = [4, 5, 6, 19, 30, 12];

            const [num0, , num3, , num6] = numbers;

            console.log(num0, num3, num6); // Logs: 4 6 30
